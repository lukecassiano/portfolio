    2. Create `components/ui/GradientBlob.tsx` as a Client Component:
       ```tsx
       'use client'

       import { motion, useReducedMotion } from 'framer-motion'
       import { cn } from '@/lib/utils'

       interface GradientBlobProps {
         /** Ordered gradient color stops, e.g. ['#F4A8C7', '#C4B5FD', '#93C5FD'] */
         colors: string[]
         /** CSS size for the blob's width & height, e.g. '60vmax'. Square. */
         size?: string
         /** Inline positioning (top/left/right/bottom/transform) for the wrapper. */
         position?: React.CSSProperties
         /** Static blur radius. Never animated. */
         blur?: string
         className?: string
       }

       export function GradientBlob({
         colors,
         size = '60vmax',
         position,
         blur = '80px',
         className,
       }: GradientBlobProps) {
         const reduce = useReducedMotion()

         const gradient = `radial-gradient(circle at 50% 50%, ${colors.join(', ')})`

         return (
           <motion.div
             aria-hidden="true"
             className={cn('pointer-events-none absolute rounded-full', className)}
             style={{
               width: size,
               height: size,
               background: gradient,
               filter: `blur(${blur})`, // STATIC — never animated
               willChange: 'transform',
               ...position,
             }}
             // Animate ONLY transform (y) and opacity. Never width/height/margin.
             initial={{ opacity: 0.85, y: 0 }}
             animate={
               reduce
                 ? { opacity: 0.85, y: 0 } // no drift under reduced-motion
                 : { opacity: [0.8, 0.95, 0.8], y: ['0%', '-4%', '0%'] }
             }
             transition={
               reduce
                 ? undefined
                 : { duration: 14, repeat: Infinity, ease: 'easeInOut' }
             }
           />
         )
       }
       ```

       Constraints to honor exactly:
       - `filter: blur(...)` is in the static `style`, NOT inside `animate`.
       - `willChange: 'transform'` present.
       - `animate` only ever changes `opacity` and `y` (transform). No width/height/scale-of-layout.
       - `useReducedMotion()` disables the drift loop.
       - `aria-hidden="true"` (decorative).

       Note on motion budget: PROJECT.md says motion is minimal — a single very slow drift loop is acceptable here as the blob's ambient behavior, gated by reduced-motion. If the executor prefers no looping at all per "no ambient animation," set the non-reduced animate to a single settle (initial opacity 0 → 0.85, y 2% → 0, no repeat) and document the choice in SUMMARY. Either way: transform+opacity only, static blur, will-change present.
